# RunPod Serverless worker dependencies (Deploy D0).
# The neuromod package + api/model_manager chain bring their own deps via `pip install -e .`
# (see repo requirements.txt); this file pins the serving-specific extras.

runpod>=1.6.0

# Model runtime. Torch is provided by the base image; these are the HF stack the
# neuromod hooks operate on (steering / attention / logits processors need Transformers).
# transformers>=4.55 is required for the gpt-oss architecture.
transformers>=4.55.0
accelerate>=0.34.0
safetensors>=0.4.0
sentencepiece>=0.2.0

# 4-bit quantization (the gpt-oss / large-model configs in model_support.py use it).
bitsandbytes>=0.43.0

# gpt-oss ships MXFP4 weights and needs its Triton kernels at runtime (Triton >= 3.4, from the
# cu128 torch the Dockerfile installs).
#
# IMPORTANT: pin kernels to [0.12, 0.15).
#   * >= 0.12: transformers' MXFP4 path needs kernels >= 0.12.0 to fetch the gpt-oss Triton
#     kernels. With an OLDER kernels it prints "MXFP4 quantization requires kernels>=0.12.0. We
#     will default to dequantizing the model to bf16" — dequantizing 120B to bf16 is ~240GB,
#     which does NOT fit the GPU (and device_map max_memory forbids CPU), so the load hangs/OOMs
#     and every request times out. The old >=0.9 floor let some builds install <0.12.
#   * < 0.15: kernels 0.15 made "a version or revision is required" a HARD error when pulling Hub
#     kernels, so the MXFP4 path fails with "Either a revision or a version must be specified."
#     See huggingface/transformers#46291.
kernels>=0.12,<0.15
triton>=3.4.0
